import { Readable } from "stream";
import * as fs from "fs";
import * as path from "path";
import { Response } from "express";
 
import { OutputModel, FileOutputModel } from "../../models/Application";
import { default as Spawn } from "./spawn";

type PipeOptions = {
  stderr?: boolean,
  stdout?: boolean,
  files?: boolean
};

type FilesCreated = FileOutputModel & {
  files: {
    fPath: string,
    fName: string
  }[]
};

 
type stdOptions = {
  stdout: {
    path: string,
    alias: string
  }
  stderr: {
    path: string,
    alias: string
  }
};
 
export abstract class PipeOutput {
  private res: Response;
  protected stream: Readable;
 
  constructor(res: Response) {
    this.res = res;
    this.stream = new Readable();
  }

  public initOutput(options: PipeOptions, model: OutputModel, spawn: Spawn, outdir: string) {
    const fOutStream = fs.createWriteStream(path.join(outdir, "stdout"));
    const fErrStream = fs.createWriteStream(path.join(outdir, "stderr"));
 
    if (options.stderr) {
      spawn.child.stderr.on("data", (data: Buffer) => {
        fErrStream.write(data);
      });
    }
    if (options.stdout) {
      spawn.child.stdout.on("data", (data: Buffer) => {
        fOutStream.write(data);
      });
    }
 
    if (options.files) {
      spawn.files.on("data", (fPath: string, fName: string) => {
        const files: FileOutputModel[] = model.files;
        const fOptions = <FilesCreated>files.find((f) => {
          return new RegExp(f.fileRgx).test(fName);
        });

        if (fOptions) {
          if (!fOptions.files) {
            fOptions.files = [];
          }
          fOptions.files.push({ fPath, fName });
        }
      });
    }

    spawn.runChild
      .then((code) => {
        fOutStream.close();
        fErrStream.close();

        this.pipeData(<FilesCreated[]>model.files, {
          stdout: {
            alias: model.stdout.alias,
            path: path.join(outdir, "stdout")
          },
          stderr: {
            alias: model.stderr.alias,
            path: path.join(outdir, "stderr")
          }
        }, code);
      })
      .catch((error) => {
        this.stream.push(`{"Fatal Error":${JSON.stringify(error)}}`);
        this.stream.push(null);
      });
  }
 
  public pipeData(files: FilesCreated[], std: stdOptions, exit_code: number) {
    this.res.setHeader("Content-Type", "application/json");
    this.stream.pipe(this.res);
    this.stream.push("{");
    this.stream.push('"' + std.stdout.alias + '":');
    this.pipeStdout(std.stdout.path);
    this.stream.push(',"' + std.stderr.alias + '":');
    this.pipeStdout(std.stderr.path);
    if (exit_code == null) {
      this.stream.push('",waring":"Process allowed execution time expired (Killed)."');
    }
    this.pipeFiles(<FilesCreated[]>files);
    this.stream.push(`,"exit_code":${exit_code}}`);
    this.stream.push(null);
  }
 
  private pipeFiles(files: FilesCreated[]) {
    for (let n = 0; n < files.length; n++) {
      const fileOutOpts = files[n];
      this.stream.push(`,"${fileOutOpts.alias}":[{`);
      for (let i = 0; fileOutOpts.files && i < fileOutOpts.files.length; i++) {
        const file = fileOutOpts.files[i];
        this.stream.push(`"${file.fName}":`);
        this.pipeFileData(path.join(file.fPath, file.fName), fileOutOpts.encoding);
 
        if (i + 1 !== fileOutOpts.files.length) {
          this.stream.push("},{");
        }
      }
      this.stream.push("}]");
    }
  }
 
  protected abstract pipeStdout(fPath: string): void;
 
  protected abstract pipeStderr(fPath: string): void;
 
  protected abstract pipeFileData(fPath: string, encoding: string): void;
}
