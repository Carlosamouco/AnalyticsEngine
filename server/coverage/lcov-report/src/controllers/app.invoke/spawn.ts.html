import * as child from "child_process";
import * as kill from "tree-kill";
import * as Promise from "bluebird";
import * as fs from "fs";
import * as path from "path";
import * as crypto from "crypto";
import * as EventEmitter from "events";
 
export type SpawnOptions = {
  output: {
    outputDir: string,
    ignoreFiles: string[]
  }
  spawn: child.SpawnOptions
};

export default class SpawnWarper {
  private _entry: string;
  private _args: string[];
  private _options: SpawnOptions;
  private _timeout: number;

  public child: child.ChildProcess;
  public runChild: Promise<any>;
  public files: EventEmitter;
 
  private _readFiles() {
    const dir = this._options.output.outputDir;
    const ignoreFiles = this._options.output.ignoreFiles;

    return new Promise((resolve, reject) => {
      fs.readdir(dir, (err, files) => {
        if (err) return reject(err);

        files.forEach((fName) => {
          if (ignoreFiles.indexOf(fName) !== -1) {
            return;
          }
          if (fs.lstatSync(path.join(dir, fName)).isFile()) {
            this.files.emit("data", dir, fName);
          }
        });

        resolve();
      });
    });
  }
 
  constructor(entry: string, args: string[], timeout: number, options: SpawnOptions) {
    this._entry = entry;
    this._args = args;
    this._timeout = timeout;
    this._options = options;
 
    this.files = new EventEmitter();
 
    this.child = child.spawn(this._entry, this._args, this._options.spawn);
 
    const timeoutFunc = setTimeout(() => { kill(this.child.pid); }, this._timeout);
 
    this.runChild = new Promise((resolve, reject) => {
      this.child.on("error", (err) => {
        clearTimeout(timeoutFunc);
        reject(err);
      });
 
      this.child.on("close", (code) => {
        clearTimeout(timeoutFunc);
        this._readFiles()
          .then(() => resolve(code))
          .catch((err) => reject(err));
      });
    });
  }
}
