import { Request, Response, NextFunction } from "express";

import { default as Spawn } from "./spawn";
import { PipeRawOutput } from "./pipe.raw";
import { ExecApp } from "./exec.app";
import { preInvokeAlgorithm } from "./pre.invoke";

type Arguments = {
  [key: string]: any
};
 
type InvokeModel = {
  app_id: string,
  version_id: string,
  args: Arguments,
  options: {
    timeout: number,
    output: {
      mode: OutputMode,
      stderr: boolean,
      stdout: boolean
    }
  }
};

enum OutputMode {
  RawJson,
  Raw
}

export async function invokeAlgorithm(req: Request, res: Response, next: NextFunction) {
  let data;

  data = await preInvokeAlgorithm(req, res, next);
 
  if (!data) {
    return;
  }
 
  const app = new ExecApp(data.call.app_id, data.algorithm);

  let process: Spawn;
 
  try {
    await app.compileArgs(data.call.args);
    process = await app.spawnProcess(data.call.options.timeout);
  }
  catch (err) {
    return next(err);
  }

  const output = data.call.options.output;

  switch (output.mode) {
    case OutputMode.RawJson:
      const pipe = new PipeRawOutput(res);
      pipe.initOutput(output, data.algorithm.output, process, app.getOutputDir());
      break;
    case OutputMode.Raw:
      //  pipeRawData(process, res, { stderr: output.stderr, stdout: output.stdout });
      break;
  }
}
 
export async function invokeAlgorithmV2(req: Request, res: Response, next: NextFunction) {
  let data;
 
  data = await preInvokeAlgorithm(req, res, next);
 
  if (!data) {
    return;
  }
 
  const app = new ExecApp(data.call.app_id, data.algorithm);
 
  try {
    await app.compileArgs(data.call.args);
  }
  catch (err) {
    return next(err);
  }
 
  app.spawnRemoteProcess(res, data.call.options.timeout)
    .catch((err) => {
      console.error(`unable to run the code in the sandbox: ${err}`);
      next(err);
    });
}
