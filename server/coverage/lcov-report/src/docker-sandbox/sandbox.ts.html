"use strict";
 
import * as Container from "./container";
import { default as PoolManager } from "./pool_manager";
import { default as Job } from "./job";
import * as fs from "fs";
//  let log         = require('winston')
 
import * as Docker from "dockerode";
import * as _ from "lodash";
 
interface SandboxOpts {
  poolSize: number;
  timeoutMs: number;
  memoryLimitMb: number;
}

interface RunOpts {
  code: string;
  timeoutMs: number;
  v3: boolean;
}
 
export const defaultOptions = {
  poolSize: 1,
  memoryLimitMb: 50,
  imageName: "docker-sandbox",
  timeoutMs: 10000
};
 
const noop = () => { };
 
export default class Sandbox {
  private options: any;
  private docker: Docker;
  private manager: PoolManager;
 
  constructor(options: SandboxOpts) {
    this.options = _.defaults(options, defaultOptions);
 
    this.options.containerLaunchOptions = {
      Image: this.options.imageName,
      NetworkDisabled: false,
      AttachStdin: false,
      AttachStdout: false,
      AttachStderr: false,
      OpenStdin: false,
      Privileged: false,
      User: "sandboxuser",
      Tty: false,
      HostConfig: {
        Memory: this.options.memoryLimitMb * 1000000,
        MemorySwap: -1,
        Privileged: false,
        CpusetCpus: "0" // only use one core
      },
      Labels: {
        __docker_sandbox: "1"
      },
      ExposedPorts: {
        "3000/tcp": {}
      }
    };
 
 
    this.docker = new Docker({
      socketPath: "/var/run/docker.sock"
    });
    this.manager = new PoolManager(this.docker, options);

    const cleanupEvents = ["beforeExit", "SIGINT"];
    const cleanupFn = this.cleanup.bind(this, true);
    cleanupEvents.map(event => {
      process.on(event, cleanupFn);
    });
 
  }
 
  /*
   * Initializes the sandbox by creating the pool of
   * containers
   */
  initialize(cb: Function) {
    this.manager.initialize(this.options.poolSize, cb);
  }

  /*
   *  (deprecated, use 'initialize')
   *  Initializes the pool and creates required containers
   */
  createPool(cb: Function) {
    this.manager.initialize(this.options.poolSize, cb);
  }

  /*
   * Runs the specifed code
   */
  run(options: RunOpts, cb: Function) {
    const code = _.isObject(options) ? options.code : options;
    const v3 = (options.v3 === true) ? true : false; // python3 switch
    const timeoutMs = _.isObject(options) && options.timeoutMs ? options.timeoutMs : this.options.timeoutMs;
 
    if (!code || !_.isString(code))
      throw new Error("Please provide the code to run as a string or an object {code: xxx}");
 
    const job = new Job(code, timeoutMs, cb, v3);
    this.manager.executeJob(job);
  }
 
 
  /*
   *  Cleanups various resources such as temporary
   *  files and docker containers
   */
  cleanup(cb: Function) {
    //  log.debug("cleaning up");
    let exit: boolean;
    if (typeof cb === "boolean") {
      cb = null;
      exit = true;
    }
    else {
      cb = cb || _.noop;
    }
 
    this.manager.cleanup((err: any) => {
      if (cb) return cb(err);
      else if (exit) process.exit();
    });
  }
 
}
